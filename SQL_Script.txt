
create table userDetails (
  userID integer NOT NULL AUTO_INCREMENT,
  userName  VARCHAR(50)
  password VARCHAR(50),
  userType VARCHAR(50),
  accountType VARCHAR(50)
  PRIMARY KEY (userID)
);
